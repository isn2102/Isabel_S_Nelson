```{r}
data("ny_noaa")
ny_noaa_df <- 
  ny_noaa %>% 
  filter(id %in% c("USW00094728", "USW00014771")) %>% 
  separate(date, into = c("year", "month", "day"), convert = TRUE) %>% 
  mutate(
    tmax = as.numeric(tmax)/10,
    tmin = as.numeric(tmin)/10,
    station = recode(
      id, 
      USW00094728 = "New_York_City", 
      USW00014771 = "Syracuse"))

month_df <-
  tibble(
    month = 1:12, 
    Month = as.factor(month.name))

ny_noaa_final <- left_join(ny_noaa_df, month_df, by = "month")
```

### Average Max Temperature

```{r}
x_axis <- list(
  nticks = 12,
  title = "Month")

y_axis <- list(
  title = "Average Maximum Temperature")

ny_noaa_df %>% 
  group_by(station, year, month) %>% 
  summarize(
    mean_tmax = round(mean(tmax, na.rm = TRUE))) %>% 
  plot_ly(
    x = ~month, y = ~mean_tmax, color = ~station, text = ~mean_tmax,
    alpha = .8, type = "scatter", mode = "lines", colors = "Dark2") %>% 
  layout(
    xaxis = x_axis, 
    yaxis = y_axis)
```

### Average Precipitation by Month

```{r}
x_axis <- list(
  nticks = 12,
  title = "Month")

y_axis <- list(
  title = "Average Precipitation")

ny_noaa_df %>% 
  group_by(station, month) %>% 
  summarize(
    mean_prcp = round(mean(prcp, na.rm = TRUE))) %>% 
  plot_ly(
    x = ~month, y = ~mean_prcp, color = ~station,
    alpha = .8, type = "bar", colors = "Dark2") %>% 
  layout(
    xaxis = x_axis, 
    yaxis = y_axis)
```

### Distribution of Precipitation - Syracuse

```{r}
x_axis <- list(
  title = "Precipitation (mm)")

y_axis <- list(
  title = "Count")

ny_noaa_df %>% 
  filter(snow != 0) %>% 
  filter(station == "Syracuse") %>% 
plot_ly(alpha = 0.8) %>%
  add_histogram(
    x = ~prcp, 
    name = "Precipitation", 
    color = "Dark Green") %>%        
  add_histogram(
    x = ~snow, 
    name = "Snowfall", 
    color = "Red") %>%    
  layout(
    barmode = "overlay", 
    xaxis = x_axis, 
    yaxis = y_axis)
```
